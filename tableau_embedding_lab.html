00

オブジェクトモデル全体像

概念

Embedding API v3 のオブジェクト階層を理解することが、すべての操作の出発点です。
<tableau-viz> Web Component もしくは TableauViz クラスが頂点となり、 そこから Workbook → Sheet (Worksheet / Dashboard) → Filter / Mark / DataSource へアクセスします。

TableauViz (= <tableau-viz> 要素) ├─ .workbook : Workbook │ ├─ .activeSheet : Sheet (Worksheet | Dashboard | Story) │ ├─ .publishedSheetsInfo : SheetInfo[] │ ├─ .activateSheetAsync(name) │ ├─ .changeParameterValueAsync(name, value) │ ├─ .getParametersAsync() │ └─ .getCustomViewsAsync() / .saveCustomViewAsync() │ ├─ Worksheet (Sheet の具体型) │ ├─ .applyFilterAsync(field, values, updateType) │ ├─ .applyRangeFilterAsync(field, rangeOpts, updateType) │ ├─ .clearFilterAsync(field) │ ├─ .getFiltersAsync() │ ├─ .selectMarksByValueAsync(selections, updateType) │ ├─ .getSelectedMarksAsync() │ ├─ .getSummaryDataReaderAsync() ← 推奨 │ ├─ .getUnderlyingTablesAsync() │ ├─ .getUnderlyingTableDataReaderAsync(tableId) │ └─ .getDataSourcesAsync() │ ├─ Dashboard (Sheet の具体型) │ ├─ .worksheets : Worksheet[] │ ├─ .objects : DashboardObject[] │ └─ .getFiltersAsync() │ ├─ Viz レベルメソッド │ ├─ .refreshDataAsync() │ ├─ .undoAsync() / .redoAsync() / .revertAllAsync() │ ├─ .pauseAutomaticUpdatesAsync() / .resumeAutomaticUpdatesAsync() │ ├─ .displayDialogAsync(dialogType) │ └─ .exportPDFAsync(sheets, options) │ └─ TableauEventType (イベント列挙) ├─ FirstInteractive ├─ MarkSelectionChanged ├─ FilterChanged ├─ ParameterChanged ├─ TabSwitched ├─ ToolbarStateChanged ├─ UrlActionTriggered └─ CustomViewLoaded / CustomViewSaved / CustomViewRemoved
💡 import について
Embedding API v3 は ES6 Module です。<script type="module"> を使い、 必要なクラス・列挙型を個別に import します。Web Component（<tableau-viz>）を HTML に書くだけなら import 不要です。
01

Web Component で基本埋め込み

基本

最もシンプルな埋め込み方法。HTML に <tableau-viz> タグを置くだけで Viz が表示されます。
属性でツールバー位置、タブの表示/非表示、初期フィルタなどを制御できます。

HTML
<!-- ① ライブラリ読み込み (ES6 Module) -->
<script type="module"
  src="https://public.tableau.com/javascripts/api/tableau.embedding.3.latest.min.js"></script>

<!-- ② Web Component を配置 -->
<tableau-viz
  id="myViz"
  src="https://public.tableau.com/views/Superstore_embedded_800x800/Overview"
  toolbar="bottom"
  hide-tabs
  width="100%"
  height="600">

  <!-- ③ 初期フィルタ (オプション) -->
  <viz-filter field="Category" value="Technology"></viz-filter>
  <viz-filter field="Region" value="West,East"></viz-filter>

  <!-- ④ 初期パラメータ (オプション) -->
  <viz-parameter name="Profit Bin Size" value="200"></viz-parameter>

</tableau-viz>
⚠️ 主要属性一覧
src — Viz URL（必須）
toolbar — "top" / "bottom" / "hidden"
hide-tabs — タブ非表示（ブール属性、値不要）
width / height — サイズ指定
device — "desktop" / "tablet" / "phone"
touch-optimize — タッチ操作最適化

02

JavaScript による動的埋め込み

基本

プログラムから TableauViz オブジェクトを生成し DOM に追加する方法です。
条件分岐で埋め込む Viz を変えたり、ユーザー操作をトリガーに Viz を切り替えたりできます。

JavaScript (ES Module)
import { TableauViz, TableauEventType }
  from 'https://public.tableau.com/javascripts/api/tableau.embedding.3.latest.min.js';

// ① TableauViz オブジェクト生成
const viz = new TableauViz();

// ② プロパティ設定
viz.src     = 'https://public.tableau.com/views/Superstore_embedded_800x800/Overview';
viz.toolbar = 'bottom';
viz.hideTabs = true;

// ③ イベント: Viz 準備完了時
viz.addEventListener(TableauEventType.FirstInteractive, () => {
  console.log('Viz ready!');
  console.log('Active sheet:', viz.workbook.activeSheet.name);
});

// ④ DOM に追加 → 自動的にレンダリング開始
document.getElementById('vizTarget').appendChild(viz);
💡 HTML属性名 vs JSプロパティ名
HTML: hide-tabs (ケバブケース) → JS: viz.hideTabs (キャメルケース)
HTML: touch-optimize → JS: viz.touchOptimize
↑ ボタンを押すと Viz が動的に埋め込まれます
03

イベントリスナー（6種）

中級

Viz 内部で起きた操作を外部の JavaScript で検知できます。 addEventListener で TableauEventType を指定します。

JavaScript
import { TableauEventType }
  from 'https://public.tableau.com/javascripts/api/tableau.embedding.3.latest.min.js';

const viz = document.getElementById('myViz');

// ① Viz 準備完了
viz.addEventListener(TableauEventType.FirstInteractive, (e) => {
  console.log('Viz is interactive');
});

// ② マーク選択変更
viz.addEventListener(TableauEventType.MarkSelectionChanged, async (e) => {
  const marks = await e.detail.getMarksAsync();
  console.log(`${marks.data[0].data.length} marks selected`);
});

// ③ フィルタ変更
viz.addEventListener(TableauEventType.FilterChanged, async (e) => {
  console.log('Filter changed:', e.detail.fieldName);
});

// ④ パラメータ変更
viz.addEventListener(TableauEventType.ParameterChanged, async (e) => {
  const param = await e.detail.getParameterAsync();
  console.log('Parameter:', param.name, '→', param.currentValue.value);
});

// ⑤ タブ切替
viz.addEventListener(TableauEventType.TabSwitched, (e) => {
  console.log('Tab switched to:', e.detail.newSheetName);
});

// ⑥ ツールバー状態変更
viz.addEventListener(TableauEventType.ToolbarStateChanged, (e) => {
  console.log('Undo possible:', e.detail.canUndo);
  console.log('Redo possible:', e.detail.canRedo);
});

04

カテゴリフィルタ操作

中級

applyFilterAsync() でカテゴリ（離散値）フィルタを操作します。
FilterUpdateType で Replace / Add / Remove / All を指定します。

JavaScript
import { FilterUpdateType }
  from 'https://public.tableau.com/javascripts/api/tableau.embedding.3.latest.min.js';

const viz = document.getElementById('myViz');
const sheet = viz.workbook.activeSheet;

// ── Replace: 指定値のみに絞る ──
await sheet.applyFilterAsync(
  "Category",
  ["Technology"],
  FilterUpdateType.Replace
);

// ── Add: 既存フィルタに値を追加 ──
await sheet.applyFilterAsync(
  "Category",
  ["Furniture"],
  FilterUpdateType.Add
);

// ── Remove: 既存フィルタから値を除外 ──
await sheet.applyFilterAsync(
  "Category",
  ["Furniture"],
  FilterUpdateType.Remove
);

// ── All: フィルタ全解除（全値表示） ──
await sheet.applyFilterAsync(
  "Category",
  [],
  FilterUpdateType.All
);

// ── clearFilterAsync: 特定フィルタをクリア ──
await sheet.clearFilterAsync("Category");

// ── 現在のフィルタ一覧取得 ──
const filters = await sheet.getFiltersAsync();
filters.forEach(f => {
  console.log(f.fieldName, f.filterType, f.appliedValues);
});
Result

05

レンジフィルタ / 日付フィルタ

中級

数値や日付の範囲でフィルタする場合は applyRangeFilterAsync() を使います。
min / max で範囲を指定します。日付は Date オブジェクトを渡します。

JavaScript
import { FilterUpdateType }
  from 'https://public.tableau.com/javascripts/api/tableau.embedding.3.latest.min.js';

const sheet = viz.workbook.activeSheet;

// ── 数値レンジフィルタ ──
await sheet.applyRangeFilterAsync(
  "Profit",
  {
    min: 0,
    max: 5000
  },
  FilterUpdateType.Replace
);

// ── 日付レンジフィルタ ──
await sheet.applyRangeFilterAsync(
  "Order Date",
  {
    min: new Date('2022-01-01'),
    max: new Date('2023-12-31')
  },
  FilterUpdateType.Replace
);

// ── HTML での初期レンジフィルタ指定 ──
// <viz-range-filter field="Profit" min="0" max="5000"></viz-range-filter>

// ── HTML での相対日付フィルタ指定 ──
// <viz-relative-date-filter
//   field="Order Date"
//   anchor-date="2023-06-01"
//   period-type="Year"
//   range-type="LastN"
//   range-n="2"
// ></viz-relative-date-filter>
06

マーク選択・取得

中級

selectMarksByValueAsync() でプログラムからマークを選択し、 getSelectedMarksAsync() で選択中のマーク情報を取得します。

JavaScript
import { SelectionUpdateType }
  from 'https://public.tableau.com/javascripts/api/tableau.embedding.3.latest.min.js';

const sheet = viz.workbook.activeSheet;

// ── 値指定でマーク選択 ──
await sheet.selectMarksByValueAsync(
  [{ fieldName: "Region", value: ["West"] }],
  SelectionUpdateType.Replace
);

// ── 範囲指定でマーク選択 ──
await sheet.selectMarksByValueAsync(
  [{
    fieldName: "AGG(Profit)",
    value: { min: 10000, max: Infinity }
  }],
  SelectionUpdateType.Replace
);

// ── 選択マークの取得 ──
const marksResult = await sheet.getSelectedMarksAsync();
const dataTable = marksResult.data[0];

// 列名の取得
const columns = dataTable.columns.map(c => c.fieldName);
console.log('Columns:', columns);

// 行データの取得
dataTable.data.forEach(row => {
  const values = row.map(cell => cell.formattedValue);
  console.log(values);
});

// ── 選択解除 ──
await sheet.clearSelectedMarksAsync();

データ取得 (Summary / Underlying)

上級

ワークシートの集計データ（Summary）やソースデータ（Underlying）を取得します。
getSummaryDataAsync() は非推奨になり、 getSummaryDataReaderAsync() が推奨されています。

JavaScript — Summary Data
const sheet = viz.workbook.activeSheet;

// ── 方法①: getSummaryDataReaderAsync (推奨, v3.4+) ──
const reader = await sheet.getSummaryDataReaderAsync();
const dataTable = await reader.getAllPagesAsync();  // 4M行以下
await reader.releaseAsync(); // リソース解放

// 列情報
console.log('Columns:', dataTable.columns.map(c => c.fieldName));

// 先頭5行を表示
dataTable.data.slice(0, 5).forEach(row => {
  console.log(row.map(cell => cell.formattedValue));
});

// ── 方法②: ページング（大量データ） ──
const reader2 = await sheet.getSummaryDataReaderAsync({ pageSize: 1000 });
for (let i = 0; i < reader2.pageCount; i++) {
  const page = await reader2.getPageAsync(i);
  console.log(`Page ${i}: ${page.data.length} rows`);
}
await reader2.releaseAsync();
JavaScript — Underlying (Full) Data
const sheet = viz.workbook.activeSheet;

// ① 論理テーブル一覧取得
const tables = await sheet.getUnderlyingTablesAsync();
console.log('Tables:', tables.map(t => t.id));

// ② 特定テーブルのデータ取得
const tableReader = await sheet.getUnderlyingTableDataReaderAsync(
  tables[0].id,
  { pageSize: 500 }
);
const fullData = await tableReader.getAllPagesAsync();
await tableReader.releaseAsync();

console.log(`Underlying rows: ${fullData.data.length}`);

// ── DataSource 経由のアクセス ──
const dataSources = await sheet.getDataSourcesAsync();
const primary = dataSources[0]; // 先頭がプライマリ
console.log('DataSource:', primary.name, primary.isExtract);

// 論理テーブル取得
const logicalTables = await primary.getLogicalTablesAsync();
const dsReader = await primary.getLogicalTableDataReaderAsync(
  logicalTables[0].id
);
const dsData = await dsReader.getAllPagesAsync();
await dsReader.releaseAsync();

Viz 制御 (Undo / Refresh / Pause)

サイズ変更・レスポンシブ

基本

Viz のサイズは HTML 属性 / CSS / JavaScript のいずれでも制御できます。
device 属性でデバイスレイアウトを切り替えられます。

HTML + CSS
<!-- 方法①: 属性で固定サイズ -->
<tableau-viz width="800" height="600" ...></tableau-viz>

<!-- 方法②: CSS で親コンテナに合わせる -->
<style>
  .viz-wrapper {
    width: 100%;
    max-width: 1200px;
    aspect-ratio: 16 / 9;
  }
  .viz-wrapper tableau-viz {
    width: 100%;
    height: 100%;
  }
</style>

<!-- 方法③: device 属性 -->
<tableau-viz device="phone" ...></tableau-viz>
JavaScript — 動的リサイズ
// ── viz 要素のサイズを JS で変更 ──
const vizElement = document.getElementById('myViz');

// ピクセル指定
vizElement.width  = 1000;
vizElement.height = 700;

// デバイスレイアウト切替
vizElement.device = 'tablet';

// ── レスポンシブ対応（ResizeObserver） ──
const observer = new ResizeObserver(entries => {
  for (const entry of entries) {
    const { width } = entry.contentRect;
    if (width < 600) {
      vizElement.device = 'phone';
    } else if (width < 1024) {
      vizElement.device = 'tablet';
    } else {
      vizElement.device = 'desktop';
    }
  }
});
observer.observe(vizElement.parentElement);

複合シナリオ: フィルタ連鎖

上級

複数の操作を async/await でチェーンし、 「シート切替 → フィルタ → マーク選択 → データ取得」を一連の流れで実行します。

JavaScript — 複合操作
async function compositeScenario() {
  const workbook = viz.workbook;

  // Step 1: シート切替
  const sheet = await workbook.activateSheetAsync("Overview");
  console.log('✓ Sheet activated:', sheet.name);

  // Step 2: カテゴリフィルタ適用
  await sheet.applyFilterAsync(
    "Category",
    ["Technology", "Furniture"],
    FilterUpdateType.Replace
  );
  console.log('✓ Category filter applied');

  // Step 3: レンジフィルタ適用
  await sheet.applyRangeFilterAsync(
    "Profit",
    { min: 0, max: 10000 },
    FilterUpdateType.Replace
  );
  console.log('✓ Profit range filter applied');

  // Step 4: マーク選択
  await sheet.selectMarksByValueAsync(
    [{ fieldName: "Region", value: ["West"] }],
    SelectionUpdateType.Replace
  );
  console.log('✓ Marks selected');

  // Step 5: 選択マークのデータ取得
  const marks = await sheet.getSelectedMarksAsync();
  const count = marks.data[0]?.data?.length || 0;
  console.log(`✓ ${count} marks retrieved`);

  // Step 6: Summary Data 取得
  const reader = await sheet.getSummaryDataReaderAsync();
  const data = await reader.getAllPagesAsync();
  await reader.releaseAsync();
  console.log(`✓ Summary: ${data.data.length} rows, ${data.columns.length} cols`);

  return { marks: count, rows: data.data.length };
}
💡 await の重要性
各操作は非同期です。await を省略すると、前の操作が完了する前に次の操作が始まり、 予期しない動作になります。特にフィルタ→マーク選択の連鎖では必ず await を使いましょう。
15

エラーハンドリング・デバッグ

上級

非同期メソッドは Promise を返すため、try/catch でエラーを捕捉します。 不正な FilterUpdateType やフィールド名の間違いは実行時エラーになります。

JavaScript — エラーハンドリング
// ── 基本パターン: try/catch ──
async function safeFilter(fieldName, values) {
  try {
    await sheet.applyFilterAsync(fieldName, values, FilterUpdateType.Replace);
    console.log('✓ Filter applied');
  } catch (err) {
    console.error(`✗ Filter error: ${err.message}`);
    // よくある原因:
    //   - フィールド名が存在しない
    //   - Viz がまだ interactive でない
    //   - FilterUpdateType が不正
  }
}

// ── FirstInteractive を待ってから操作 ──
function waitForViz(vizElement) {
  return new Promise((resolve) => {
    vizElement.addEventListener(
      TableauEventType.FirstInteractive,
      () => resolve(vizElement),
      { once: true }
    );
  });
}

// 使用例
const readyViz = await waitForViz(document.getElementById('myViz'));
await safeFilter('Category', ['Technology']);

// ── デバッグ: Viz の状態確認 ──
function debugViz(viz) {
  console.group('Viz Debug Info');
  console.log('src:', viz.src);
  console.log('workbook:', viz.workbook?.name);
  console.log('activeSheet:', viz.workbook?.activeSheet?.name);
  console.log('sheetType:', viz.workbook?.activeSheet?.sheetType);
  console.log('autoUpdate paused:', viz.automaticUpdatesArePaused);

  const sheets = viz.workbook?.publishedSheetsInfo || [];
  console.log('published sheets:', sheets.map(s => s.name));
  console.groupEnd();
}
⚠️ よくあるエラーと対策
「Cannot read properties of undefined」 — Viz がまだ準備完了していません。FirstInteractive イベントを待ちましょう。
「Invalid filter field」 — フィールド名がワークブックに存在しません。getFiltersAsync() で確認してください。
「Invalid FilterUpdateType」 — 文字列リテラルではなく enum（FilterUpdateType.Replace）を使ってください。
