00

for 文（カウンタ型）

基本

最も基本的なループ。初期化; 条件; 更新 の3式で制御する。 break / continue / ラベル付きジャンプがすべて使える。

JavaScript
// 基本: 0〜4 を出力
for (let i = 0; i < 5; i++) {
  console.log(`i = ${i}`);
}

// 逆順ループ
for (let i = 4; i >= 0; i--) {
  console.log(`逆順: ${i}`);
}

// break でループ脱出
for (let i = 0; i < 10; i++) {
  if (i === 3) break;
  console.log(`break前: ${i}`);
}

// continue で偶数スキップ
for (let i = 0; i < 6; i++) {
  if (i % 2 === 0) continue;
  console.log(`奇数のみ: ${i}`);
}

while 文

基本

条件が true の間ループし続ける。条件が最初から偽なら1度も実行されない。 カウンタ更新を忘れると無限ループになるので要注意。

JavaScript
let count = 0;
while (count < 5) {
  console.log(`count = ${count}`);
  count++;
}

// 条件が最初から偽 → 実行されない
let x = 10;
while (x < 5) {
  console.log('これは出力されない');
  x++;
}
console.log(`while終了後 x = ${x}`);
02

do...while 文

基本

while と異なり、最低1回は必ず実行される。条件判定がループ末尾にある後判定ループ。

JavaScript
// 条件が偽でも1回は実行される
let n = 10;
do {
  console.log(`do...while: n = ${n}`);
  n++;
} while (n < 5);

console.log(`→ 最終 n = ${n} (1回だけ実行された)`);

// 通常の繰返し
let m = 0;
do {
  console.log(`ループ中: m = ${m}`);
  m++;
} while (m < 3);
03

for...in（プロパティ列挙）

中級

オブジェクトの列挙可能なプロパティ名（キー）を反復する。 プロトタイプチェーン上のプロパティも列挙されるため、hasOwnProperty でのガードが推奨。 配列には使わないのがベストプラクティス（インデックスが文字列で返り、順序が保証されない場合がある）。

JavaScript
const user = { name: 'Tom', age: 40, city: 'Osaka' };

// キーを列挙
for (const key in user) {
  console.log(`${key}: ${user[key]}`);
}

// ⚠ 配列に for...in → インデックスが「文字列」
const arr = ['a', 'b', 'c'];
for (const idx in arr) {
  console.log(`idx="${idx}" (型: ${typeof idx})`);
}

// hasOwnProperty でプロトタイプ除外
function Animal(name) { this.name = name; }
Animal.prototype.legs = 4;
const dog = new Animal('Pochi');

for (const key in dog) {
  const own = dog.hasOwnProperty(key) ? 'own' : 'prototype';
  console.log(`${key} = ${dog[key]} [${own}]`);
}
04

for...of（イテラブル反復）

中級

Symbol.iterator を持つイテラブル（配列、文字列、Map、Set、NodeList 等）の値を反復。 for...in と違い、キーではなく値が返る。break / continue が使える。

JavaScript
// 配列の値を反復
const fruits = ['🍎', '🍌', '🍇'];
for (const fruit of fruits) {
  console.log(fruit);
}

// 文字列の文字を反復（サロゲートペア安全）
for (const ch of '大阪🏯') {
  console.log(ch);
}

// break で途中脱出
for (const n of [10, 20, 30, 40]) {
  if (n > 25) break;
  console.log(`値: ${n}`);
}

// ⚠ 普通のオブジェクトは iterable ではない → エラー
// for (const v of {a:1}) {} // TypeError!
console.log('※ 普通のオブジェクトは for...of 不可（TypeError）');
05

forEach（配列メソッド）

基本

Array.prototype.forEach はコールバック関数を各要素に適用する。 break / continue は使えない。return は現在の反復をスキップするだけ（ループ全体は止まらない）。 戻り値は常に undefined。

JavaScript
const colors = ['red', 'green', 'blue'];

// 基本: (要素, インデックス, 配列本体)
colors.forEach((color, i, arr) => {
  console.log(`[${i}] ${color}  (配列長: ${arr.length})`);
});

// return は continue 相当（ループは止まらない）
[1,2,3,4,5].forEach(n => {
  if (n === 3) return; // 3 だけスキップ
  console.log(`n = ${n}`);
});

// ⚠ break は使えない（構文エラー）
console.log('※ forEach 内で break は SyntaxError');

// 戻り値は undefined
const result = [1,2].forEach(x => x * 2);
console.log(`forEach戻り値: ${result}`);
06

map（変換して新配列を返す）

基本

各要素にコールバックを適用し、戻り値で新しい配列を生成する。 元の配列は変更しない（非破壊）。forEach との最大の違いは「戻り値がある」こと。

JavaScript
const nums = [1, 2, 3, 4];

// 2倍にした新配列
const doubled = nums.map(n => n * 2);
console.log('元配列:', nums);
console.log('map結果:', doubled);

// オブジェクト配列の変換
const users = [
  { name: 'Alice', score: 80 },
  { name: 'Bob',   score: 95 },
];
const names = users.map(u => u.name);
console.log('名前だけ抽出:', names);

// インデックスも使える
const indexed = ['a','b','c'].map((v, i) => `${i}:${v}`);
console.log(indexed);
07

filter（条件で抽出）

基本

コールバックが true を返す要素だけを集めた新しい配列を返す。 元の配列は変更されない。条件に合う要素がなければ空配列 []。

JavaScript
const scores = [45, 82, 73, 91, 58, 66];

// 70以上を抽出
const passed = scores.filter(s => s >= 70);
console.log('合格者:', passed);

// 条件不一致 → 空配列
const perfect = scores.filter(s => s === 100);
console.log('満点者:', perfect);

// falsy 除去テクニック
const mixed = [0, '', null, 'hello', undefined, 42, false];
const truthy = mixed.filter(Boolean);
console.log('truthy のみ:', truthy);
08

reduce（集約・畳み込み）

中級

配列を単一の値に集約する。アキュムレータ（累積値）と現在の要素を受け取り、最終結果を返す。 初期値を省略すると先頭要素が初期値になるが、空配列で TypeError が出るため常に初期値を渡すのが安全。

🔰 reduce のデータフロー
  初期値: 0
    ↓
  [10, 20, 30]
    │
    ├─ acc=0,  cur=10  → return 10
    ├─ acc=10, cur=20  → return 30
    └─ acc=30, cur=30  → return 60  ← 最終結果
JavaScript
// 合計
const sum = [10, 20, 30].reduce((acc, cur) => {
  console.log(`  acc=${acc}, cur=${cur}`);
  return acc + cur;
}, 0);
console.log(`合計: ${sum}`);

// オブジェクトへの集約（グルーピング）
const items = ['apple','banana','apple','cherry','banana','apple'];
const counts = items.reduce((obj, item) => {
  obj[item] = (obj[item] || 0) + 1;
  return obj;
}, {});
console.log('出現回数:', JSON.stringify(counts));

// 最大値
const max = [3,7,1,9,4].reduce((a, b) => a > b ? a : b);
console.log(`最大値: ${max}`);

some / every（存在・全称判定）

基本

some: 1つでも条件を満たせば true（短絡評価で即停止）。 every: 全要素が条件を満たせば true（1つでも偽で即停止）。

JavaScript
const ages = [15, 22, 17, 30];

// some: 1人でも成人がいるか？
const hasAdult = ages.some(a => {
  console.log(`  some チェック: ${a}`);
  return a >= 18;
});
console.log(`成人がいる: ${hasAdult}`);

// every: 全員が成人か？
const allAdult = ages.every(a => {
  console.log(`  every チェック: ${a}`);
  return a >= 18;
});
console.log(`全員成人: ${allAdult}`);

// 空配列の挙動
console.log(`空.some:  ${[].some(() => true)}`);  // false
console.log(`空.every: ${[].every(() => false)}`); // true（空真）

find / findIndex（最初の一致を検索）

基本

find: 条件を満たす最初の要素を返す（なければ undefined）。 findIndex: 条件を満たす最初のインデックスを返す（なければ -1）。

JavaScript
const products = [
  { id: 1, name: 'ペン',    price: 100 },
  { id: 2, name: 'ノート',  price: 250 },
  { id: 3, name: '消しゴム', price: 80  },
];

// find: 最初に200円以上の商品
const found = products.find(p => p.price >= 200);
console.log('find:', found);

// findIndex: インデックスを取得
const idx = products.findIndex(p => p.name === '消しゴム');
console.log(`findIndex: ${idx}`);

// 見つからない場合
const notFound = products.find(p => p.price > 1000);
console.log(`見つからない: ${notFound}`); // undefined

// findLast / findLastIndex（ES2023）
const nums = [1,2,3,4,5];
const lastEven = nums.findLast(n => n % 2 === 0);
console.log(`findLast (偶数): ${lastEven}`);

flatMap（展開＋変換）

中級

map + flat(1) を1ステップで行う。 各要素から配列を返すと結果を1段階フラットにする。要素の除去（空配列を返す）にも使える。

JavaScript
// 文を単語に分割して1次元化
const sentences = ['hello world', 'foo bar baz'];
const words = sentences.flatMap(s => s.split(' '));
console.log(words);

// 要素の除去（偶数だけ残す）
const nums = [1,2,3,4,5];
const evens = nums.flatMap(n => n % 2 === 0 ? [n] : []);
console.log('偶数のみ:', evens);

// 1対多の展開
const data = [{ tag: 'a', vals: [1,2] }, { tag: 'b', vals: [3] }];
const flat = data.flatMap(d => d.vals.map(v => ({ tag: d.tag, val: v })));
console.log(JSON.stringify(flat));

entries / keys / values

中級

配列・オブジェクトの entries(), keys(), values() で イテレータを取得し、for...of と組み合わせる。 オブジェクトには Object.entries() 等の静的メソッドを使う。

JavaScript
// 配列の entries()
const arr = ['🍎', '🍌', '🍇'];
for (const [i, v] of arr.entries()) {
  console.log(`[${i}] ${v}`);
}

// オブジェクトの Object.entries()
const config = { host: 'localhost', port: 3000, debug: true };
for (const [key, val] of Object.entries(config)) {
  console.log(`${key} → ${val}`);
}

// Object.keys / Object.values
console.log('keys:', Object.keys(config));
console.log('values:', Object.values(config));

Map / Set の反復

中級

Map と Set はイテラブルなので for...of と forEach の両方が使える。 Map の for...of では [key, value] の分割代入が便利。

JavaScript
// Map の反復
const map = new Map([
  ['name', 'Tom'],
  ['lang', 'JavaScript'],
  ['city', 'Osaka'],
]);

for (const [k, v] of map) {
  console.log(`Map: ${k} = ${v}`);
}

map.forEach((val, key) => {
  console.log(`Map.forEach: ${key} → ${val}`);
});

// Set の反復
const set = new Set([1, 2, 3, 2, 1]);
console.log(`Set サイズ: ${set.size}`);
for (const v of set) {
  console.log(`Set: ${v}`);
}

ジェネレータと for...of

上級

function* で定義するジェネレータは yield で値を逐次返すイテラブル。 for...of で消費でき、無限シーケンスも安全に扱える（消費側で break する）。

JavaScript
// 有限ジェネレータ
function* range(start, end) {
  for (let i = start; i < end; i++) {
    yield i;
  }
}

for (const n of range(3, 7)) {
  console.log(`range: ${n}`);
}

// 無限ジェネレータ（break で安全に停止）
function* fibonacci() {
  let a = 0, b = 1;
  while (true) {
    yield a;
    [a, b] = [b, a + b];
  }
}

let count = 0;
for (const fib of fibonacci()) {
  if (count++ >= 10) break; // 安全に停止
  console.log(`fib[${count-1}] = ${fib}`);
}

// スプレッドで配列化（有限のみ！）
const arr = [...range(0, 5)];
console.log('スプレッド:', arr);

for await...of（非同期イテレータ）

上級

async function* で作る非同期ジェネレータや、Promiseの配列を for await...of で順次消費する。API のページング処理等で活躍。

JavaScript
// 非同期ジェネレータ
async function* asyncRange(start, end) {
  for (let i = start; i < end; i++) {
    await new Promise(r => setTimeout(r, 100));
    yield i;
  }
}

(async () => {
  for await (const n of asyncRange(0, 4)) {
    console.log(`async: ${n}`);
  }

  // Promise配列を for await...of で逐次処理
  const promises = [
    Promise.resolve('A'),
    Promise.resolve('B'),
    Promise.resolve('C'),
  ];
  for await (const val of promises) {
    console.log(`Promise値: ${val}`);
  }
  console.log('非同期処理完了');
})();

再帰によるループ代替

上級

ループの代わりに関数が自分自身を呼ぶパターン。ツリー走査やネスト構造の処理で必須。 終了条件（ベースケース）を必ず定義しないとスタックオーバーフローになる。

JavaScript
// 階乗（n!）
function factorial(n) {
  if (n <= 1) return 1; // ベースケース
  return n * factorial(n - 1);
}
console.log(`5! = ${factorial(5)}`);

// ネストオブジェクトのフラット化
function flattenObj(obj, prefix = '') {
  const result = {};
  for (const [k, v] of Object.entries(obj)) {
    const key = prefix ? `${prefix}.${k}` : k;
    if (v && typeof v === 'object' && !Array.isArray(v)) {
      Object.assign(result, flattenObj(v, key));
    } else {
      result[key] = v;
    }
  }
  return result;
}

const nested = { a: { b: { c: 1 }, d: 2 }, e: 3 };
console.log(JSON.stringify(flattenObj(nested)));

// 木構造の走査
const tree = {
  name: 'root',
  children: [
    { name: 'A', children: [{ name: 'A1', children: [] }] },
    { name: 'B', children: [] },
  ]
};
function walk(node, depth = 0) {
  console.log(`${'  '.repeat(depth)}└ ${node.name}`);
  node.children.forEach(c => walk(c, depth + 1));
}
walk(tree);

全パターン比較表

まとめ

全ループ手法の特徴を一覧で比較。使い分けの判断基準にどうぞ。

構文 対象 返すもの break continue 戻り値 主な用途
for 何でも（手動制御） — ✓ ✓ なし カウンタ制御、インデックス操作
while 何でも（条件制御） — ✓ ✓ なし 回数不定の繰返し
do...while 何でも（後判定） — ✓ ✓ なし 最低1回の実行保証
for...in オブジェクト キー（文字列） ✓ ✓ なし プロパティ列挙
for...of イテラブル 値 ✓ ✓ なし 配列/Map/Set/文字列の値反復
forEach 配列/Map/Set 値, index, 元配列 ✗ ✗ undefined 副作用ベースの処理
map 配列 値, index, 元配列 ✗ ✗ 新配列 変換（1対1）
filter 配列 値, index, 元配列 ✗ ✗ 新配列 条件抽出
reduce 配列 acc, cur, index ✗ ✗ 任意の値 集約・畳み込み
some 配列 値, index, 元配列 短絡停止 ✗ boolean 存在判定（1つでもtrue）
every 配列 値, index, 元配列 短絡停止 ✗ boolean 全称判定（全てtrue）
find 配列 値, index, 元配列 短絡停止 ✗ 要素 or undefined 最初の一致を取得
flatMap 配列 値, index, 元配列 ✗ ✗ 新配列（1段flat） 展開＋変換・要素除去
for await...of 非同期イテラブル Promiseの解決値 ✓ ✓ なし 非同期ストリーム
再帰 何でも — return — 任意 ツリー走査・ネスト構造
🔰 どれを使うか迷ったら
  配列を変換したい？        → map
  配列を絞りたい？          → filter
  配列を1つの値にしたい？   → reduce
  条件に合うものを探したい？ → find
  1つでも/全部 を判定？     → some / every
  副作用だけ？(何も返さない) → forEach
  途中で止めたい？          → for...of or for
  オブジェクトのキーが欲しい？→ for...in or Object.keys
  非同期を順次処理？        → for await...of
  ネスト構造の走査？        → 再帰
